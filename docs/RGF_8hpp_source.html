136  matrix_size = ns * nt + nd;

138 
139  if (nd > 0)
140  NBlock = nt + 1;
141  else
142  NBlock = nt;
143  Bmin = new size_t[NBlock];
144  Bmax = new size_t[NBlock];
145  // Calculate the start- and end index of each block
146  for (size_t i = 0; i < nt; i++) {
147  Bmin[i] = i * ns;
148  Bmax[i] = (i + 1) * ns;
149  }
150  if (nd > 0) {
151  Bmin[nt] = nt * ns;
152  Bmax[nt] = nt * ns + nd;
153  }
154 
155  // TODO: what are we doing here ?
156  // blocks with lda = 2*ns + nd
157  diag_pos = new size_t[matrix_size];
158  size_t IB;
159  for (IB = 0; IB < nt - 1; IB++) {
160  for (size_t i = 0; i < ns; i++) {
161  diag_pos[IB * ns + i] =
162  IB * ns * (2 * ns + nd) + i * (2 * ns + nd + 1);
163  }
164  }
165  // last block with lda = ns + nd
166  IB = nt - 1;
167  for (size_t i = 0; i < ns; i++) {
168  diag_pos[IB * ns + i] = IB * ns * (2 * ns + nd) + i * (ns + nd + 1);
169  }
170  // dense block with lda = ns + nd
171  if (nd > 0) {
172  IB = nt;
173  for (size_t i = 0; i < nd; i++) {
174  diag_pos[nt * ns + i] =
175  (nt - 1) * ns * (2 * ns + nd) + ns * (ns + nd) + i * (nd + 1);
176  }
177  }

454 template <class T> double RGF<T>::SecondStageSolve(size_t nrhs) {
455  int info;
456  size_t IB;
457  size_t NR, NM, NP;
458  T ONE = f_one();
459 
460  // Forward pass
461  NR = Bmax[0] - Bmin[0];
462 
463  // counting FLOPS
464  double flops = 0;
465 
466  // rj dtrsm LLNN MF[0,0] rhs[0]
467  // solve for first diagonal block
468  c_ttrsm('L', 'L', 'N', 'N', NR, nrhs, ONE, &MF[mf_block_index(0, 0)],
469  mf_block_lda(0, 0), &rhs[Bmin[0]], matrix_size);
470  flops += 2.0 * NR * NR * nrhs;
471 
472  // rj IB = 1; IB < NBlock; IB++
473  for (IB = 1; IB < matrix_nt; IB++) {
474  NR = Bmax[IB] - Bmin[IB];
475  NM = Bmax[IB - 1] - Bmin[IB - 1];
476 
477  // rj dgemm NN M[IB,IB-1] rhs[IB-1] rhs[IB]
478  c_tgemm('N', 'N', NR, nrhs, NM, -ONE, &MF[mf_block_index(IB, IB - 1)],
479  mf_block_lda(IB, IB - 1), &rhs[Bmin[IB - 1]], matrix_size, ONE,
480  &rhs[Bmin[IB]], matrix_size);
481  flops += 2.0 * NR * nrhs * NM;
482  // rj dtrsm LLNN MF[IB,IB] rhs[IB]
483  c_ttrsm('L', 'L', 'N', 'N', NR, nrhs, ONE, &MF[mf_block_index(IB, IB)],
484  mf_block_lda(IB, IB), &rhs[Bmin[IB]], matrix_size);
485  flops += 2.0 * NR * NR * nrhs;
486  }
487 
488  if (matrix_nd > 0) {
489  IB = NBlock - 1;
490  NR = Bmax[IB] - Bmin[IB];
491  NM = Bmax[IB - 1] - Bmin[IB - 1];
492 
493  // rj dgemm NN M[IB,IB-1] rhs[IB-1] rhs[IB]
494  for (size_t i = 0; i < NBlock - 1; i++) {
495  c_tgemm('N', 'N', NR, nrhs, NM, -ONE, &MF[mf_dense_block_index(i)],
496  mf_dense_block_lda(i), &rhs[Bmin[i]], matrix_size, ONE,
497  &rhs[Bmin[IB]], matrix_size);
498  flops += 2.0 * NR * nrhs * NM;
499  }
500  // rj dtrsm LLNN MF[IB,IB] rhs[IB]
501  c_ttrsm('L', 'L', 'N', 'N', NR, nrhs, ONE, &MF[mf_block_index(IB, IB)],
502  mf_block_lda(IB, IB), &rhs[Bmin[IB]], matrix_size);
503  flops += 2.0 * NR * NR * nrhs;
504  }
505 
506  // Backward pass
507  if (matrix_nd > 0) {
508  IB = NBlock - 1;
509  NR = Bmax[IB] - Bmin[IB];
510  NM = Bmax[IB - 1] - Bmin[IB - 1];
511 
512  // rj dtrsm LLTN MF[NBlock-1,NBlock-1] rhs[NBlock-1]
513  c_ttrsm('L', 'L', 'T', 'N', NR, nrhs, ONE, &MF[mf_block_index(IB, IB)],
514  mf_block_lda(IB, IB), &rhs[Bmin[IB]], matrix_size);
515  flops += 2.0 * NR * NR * nrhs;
516  // rj dgemm TN M[IB,IB-1] rhs[IB] rhs[IB-1]
517  for (size_t i = 0; i < NBlock - 1; i++) {
518  c_tgemm('T', 'N', NM, nrhs, NR, -ONE, &MF[mf_dense_block_index(i)],
519  mf_dense_block_lda(i), &rhs[Bmin[IB]], matrix_size, ONE,
520  &rhs[Bmin[i]], matrix_size);
521  flops += 2.0 * NM * nrhs * NR;
522  }
523  }
524 
525  for (IB = matrix_nt - 1; IB > 0; IB--) {
526  NR = Bmax[IB] - Bmin[IB];
527  NM = Bmax[IB - 1] - Bmin[IB - 1];
528 
529  // rj dtrsm LLTN MF[IB,IB] rhs[IB]
530  c_ttrsm('L', 'L', 'T', 'N', NR, nrhs, ONE, &MF[mf_block_index(IB, IB)],
531  mf_block_lda(IB, IB), &rhs[Bmin[IB]], matrix_size);
532  flops += 2.0 * NR * NR * nrhs;
533  // rj dgemm TN M[IB,IB-1] rhs[IB] rhs[IB-1]
534  c_tgemm('T', 'N', NM, nrhs, NR, -ONE, &MF[mf_block_index(IB, IB - 1)],
535  mf_block_lda(IB, IB - 1), &rhs[Bmin[IB]], matrix_size, ONE,
536  &rhs[Bmin[IB - 1]], matrix_size);
537  flops += 2.0 * NM * nrhs * NR;
538  }
539 
540  IB = 0;
541  NR = Bmax[IB] - Bmin[IB];
542 
543  // rj dtrsm LLTN MF[NBlock-1,NBlock-1] rhs[NBlock-1]
544  c_ttrsm('L', 'L', 'T', 'N', NR, nrhs, ONE, &MF[mf_block_index(IB, IB)],
545  mf_block_lda(IB, IB), &rhs[Bmin[IB]], matrix_size);
546  flops += 2.0 * NR * NR * nrhs;
547 
548  return flops;
549 }
